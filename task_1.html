def proverka(chech_data):
    dict_1 = {}
    for item in chech_data.columns:
        dict_1[item] = chech_data[item].isnull().sum()
    dict_1 = dict(sorted(dict_1.items(), key=lambda x: x[1], reverse=True))
    nancute = []
    for key,item in dict_1.items():
        print('''  {}  {}  '''.format(key, item))
        
            
def proverka_zero(chech_data):
    dict_1 = {}
    for item in chech_data.columns:
        dict_1[item] = chech_data[item].isin([0]).sum()
    dict_1 = dict(sorted(dict_1.items(), key=lambda x: x[1], reverse=True))
    nancute = []
    for key,item in dict_1.items():
        print('''  {}  - ,   0 : {} '''.format(key, item))

def proverka_minus(chech_data):
    dict_1 = {}
    for item in chech_data.columns:
        try:
            dict_1[item] = len(chech_data[chech_data[item] < 0])
        except:
            pass
    dict_1 = dict(sorted(dict_1.items(), key=lambda x: x[1], reverse=True))

        print('''  {}  {}  '''.format(key, item))
